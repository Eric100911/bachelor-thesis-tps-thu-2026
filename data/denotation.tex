\begin{denotation}[3cm]
  \item[LHC] 大型强子对撞机（Large Hadron Collider）
  \item[CMS] 紧凑缪子螺线管（Compact Muon Solenoid）
  \item[CERN] 欧洲核子研究中心（European Organization for Nuclear Research）
  \item[QCD] 量子色动力学（Quantum Chromodynamics）
  \item[pQCD] 微扰量子色动力学（Perturbative QCD）
  \item[NRQCD] 非相对论量子色动力学（Nonrelativistic QCD）
  \item[PDF] 部分子分布函数（Parton Distribution Function）
  \item[dPDF] 双部分子分布函数（Double Parton Distribution Function）
  \item[DGLAP] Dokshitzer--Gribov--Lipatov--Altarelli--Parisi 演化方程
  \item[SPS] 单部分子散射（Single-Parton Scattering）
  \item[DPS] 双部分子散射（Double-Parton Scattering）
  \item[TPS] 三重部分子散射（Triple-Parton Scattering）
  \item[MPS] 多重部分子散射（Multiple-Parton Scattering）
  \item[MPI] 多重部分子相互作用（Multiple Parton Interaction）
  \item[CSM] 色单态模型（Color-Singlet Model）
  \item[COM] 色八重态机制（Color-Octet Mechanism）
  \item[CEM] 色蒸发模型（Color-Evaporation Model）
  \item[SDC] 短程系数（Short-Distance Coefficient）
  \item[LDME] 长程矩阵元（Long-Distance Matrix Element）
  \item[MC] 蒙特卡罗（Monte Carlo）
  \item[HLT] 高级触发（High-Level Trigger）
  \item[L1T] 第一级触发（Level-1 Trigger）
  \item[DT] 漂移管（Drift Tube）
  \item[CSC] 阴极条室（Cathode Strip Chamber）
  \item[RPC] 阻性板室（Resistive Plate Chamber）
  \item[TIB] 径迹探测器内桶（Tracker Inner Barrel）
  \item[TOB] 径迹探测器外桶（Tracker Outer Barrel）
  \item[TID] 径迹探测器内盘（Tracker Inner Disks）
  \item[TEC] 径迹探测器端盖（Tracker Endcaps）
  \item[PF] 粒子流（Particle Flow）
  \item[PV] 初级顶点（Primary Vertex）
  \item[CTF] 组合径迹寻找器（Combinatorial Track Finder）
  \item[DA] 确定性退火（Deterministic Annealing）
  \item[AVF] 自适应顶点拟合器（Adaptive Vertex Fitter）
  \item[DSCB] 双面 Crystal Ball（Double-Sided Crystal Ball）
  \item[UML] 无分箱最大似然（Unbinned Maximum Likelihood）
  \item[PDG] 粒子数据组（Particle Data Group）
  \item[sPlot] 信号权重分离技术（Statistical Plotting Tool）~\cite{Pivk:2004ty}
  \item[$\sigma_{\mathrm{eff}}$] 有效截面（Effective Cross Section）
\end{denotation}
